# install

### using apt

```bash
sudo apt-get install ros-noetic-sophus
```

### from source

```Bash
git clone https://github.com/strasdat/Sophus.git
cd sophus
mkdir build
cd build
cmake ..
make
sudo make install
```

# check version

### using pkg-config

```bash
pkg-config --modversion Sophus
# or
pkg-config --modversion sophus
```

### using CMake package file

```bash
cat /usr/local/lib/cmake/Sophus/SophusConfigVersion.cmake | grep VERSION
```

### Check header file

```sh
grep SOPHUS_VERSION /usr/local/include/sophus/common.hpp
```

Output example:

```bash
#define SOPHUS_VERSION_MAJOR 1
#define SOPHUS_VERSION_MINOR 22
#define SOPHUS_VERSION_PATCH 10
```

### Use CMake to print version

```cmake
find_package(Sophus REQUIRED)
message(STATUS "Sophus version: ${Sophus_VERSION}")
```
